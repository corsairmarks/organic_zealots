opinion_organic_zealots_vs_machines = {
	trigger = {
		has_valid_civic = civic_organic_zealots
		from = {
			is_robot_empire = yes
			is_homicidal = no
		}
	}
	opinion = {
		base = -1000
		modifier = {
			add = -500
			OR = {
				has_valid_civic = civic_machine_assimilator
				has_valid_civic = civic_machine_terminator
				has_valid_civic = civic_machine_servitor
			}
		}
	}
}
