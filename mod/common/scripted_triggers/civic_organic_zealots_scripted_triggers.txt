	custom_tooltip = {

			is_same_species = from
			NOT = { has_valid_civic = civic_fanatic_purifiers	}
		}
	}
	custom_tooltip = {
		fail_text = "requires_actor_not_organic_zealots"
		OR = {
			is_same_species = from

			AND = {

	}
