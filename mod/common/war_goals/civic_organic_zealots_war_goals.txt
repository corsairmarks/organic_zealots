wg_force_policy_outlaw_robots_and_ai = {
	casus_belli = cb_impose_policy
	hide_if_no_cb = no
	release_occupied_systems_on_status_quo = yes
	allowed_peace_offers = {
		status_quo
		surrender
		demand_surrender
	}
	surrender_acceptance = -100
	war_exhaustion = 0.75
	potential = {
		NOT = { has_policy_flag = no_wars }
		has_valid_civic = civic_organic_zealots
		is_robot_empire = no
		exists = from
		has_total_war_cb = no
		from = { is_valid_target_to_outlaw_cybernetics_or_robot_pops_or_ai = yes }
		NOT = { is_in_federation_with = from }
		NAND = {
			is_galactic_emperor = yes
			from = { is_galactic_community_member = yes }
		}
	}
	possible = {
		hidden_trigger = { exists = from }
		custom_tooltip = {
			fail_text = has_claimed_all_systems
			from = {
				is_country_type = default
				any_owned_planet = {
					is_colony = yes
					root = {
						NOT = { has_claim = prev.solar_system }
					}
				}
			}
		}
		custom_tooltip = {
			fail_text = cannot_use_against_emperor
			from = { is_galactic_emperor = no }
		}
		custom_tooltip = {
			fail_text = pax_galactica_active
			NAND = {
				is_active_resolution = resolution_emperor_pax_galactica
				is_galactic_community_member = yes
				from = { is_galactic_community_member = yes }
			}
		}
	}
	on_status_quo = {
		if = {
			limit = {
				exists = last_created_country
				last_created_country = { has_country_flag = released_empire }
			}
			last_created_country = {
				set_origin_effect = yes
				if = {
					limit = {
						prev = { is_megacorp = yes }
						is_megacorp = yes
					}
					set_merchant_government_effect = yes
				}
				force_policy_outlaw_robots_and_ai_effect = yes
				force_policy_liberation_effect = yes
			}
			from = {
				add_threat = {
					who = root
					amount = 0.10
				}
			}
		}
	}
	on_accept = {
		from = {
			custom_tooltip = wg_force_policy_outlaw_robots_and_ai_accept
			if = {
				limit = { from = { exists = ruler } }
				custom_tooltip = wg_force_policy_outlaw_robots_and_ai_accept_2
			}
			every_subject = {
				limit = {
					is_robot_empire = no
					any_war = {
						is_war_participant = {
							who = prev
							side = prevprev
						}
					}
				}
				custom_tooltip = wg_force_policy_outlaw_robots_and_ai_accept_subject
			}
		}
		hidden_effect = {
			from = {
				force_policy_outlaw_robots_and_ai_effect = yes
				force_policy_outlaw_robots_and_ai_ruler_abdicates_effect = yes
				force_policy_liberation_effect = yes
				every_subject = {
					limit = {
						is_robot_empire = no
						any_war = {
							is_war_participant = {
								who = prev
								side = prevprev
							}
						}
					}
					force_policy_outlaw_robots_and_ai_effect = yes
					force_policy_outlaw_robots_and_ai_ruler_abdicates_effect = yes
					force_policy_liberation_effect = yes
				}
				add_threat = {
					who = root
					amount = 0.10
				}
			}
		}
	}
	ai_weight = {
		weight = 5
		modifier = {
			factor = 100
			has_valid_civic = civic_organic_zealots
		}
	}
}
