# Modified: robotic species cannot be tech assimilated, restrict tech assimilation (with Civic: Organic Zealots)
living_standard_tech_assimilation = {
	pop_modifier = {
		pop_happiness = -0.25
	}
	potential = {
		OR = {
			has_utopia = yes
			has_synthethic_dawn = yes
		}
		exists = from
		from = {
			OR = {
				AND = {
					is_machine_empire = yes
					has_active_tradition = tr_synthetics_adopt_machine
					prev = {
						OR = {
							has_trait = trait_mechanical
							has_trait = trait_machine_unit
							has_trait = trait_cybernetic
						}
					}
				}
				has_country_flag = synthetic_empire
			}
		}
	}
	allow = {
		hidden_trigger = { exists = from }
		custom_tooltip = {
			fail_text = SPECIES_MUST_BE_ASSIMILATING
			has_citizenship_type = {
				type = citizenship_assimilation
				country = from
			}
		}
		custom_tooltip = {
			fail_text = lost_robot_policy_war_fail
			from = { NOT = { has_country_flag = lost_robot_policy_war } }
		}
		custom_tooltip = {
			fail_text = is_organic_zealots_fail
			from = { NOT = { has_valid_civic = civic_organic_zealots } }
		}
		custom_tooltip = {
			fail_text = is_overlord_organic_zealots_fail
			from = {
				NAND = {
					is_subject = yes
					exists = overlord
					overlord = { has_valid_civic = civic_organic_zealots }
				}
			}
		}
	}
	ai_will_do = {
		factor = 100
	}
}
