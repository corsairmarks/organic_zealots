tr_synthetics_synthetic_age = {
	possible = {
		if = {
			limit = {
				OR = {
					has_technology = tech_synthetic_leaders
					is_machine_empire = yes
				}
			}
			always = yes
		}
		else = {
			has_technology = tech_synthetic_leaders
		}
		custom_tooltip = {
			fail_text = "lost_robot_policy_war_fail"
			NOT = { has_country_flag = lost_robot_policy_war }
		}
		custom_tooltip = {
			fail_text = "civic_tooltip_not_organic_zealots_fail"
			NOT = { has_valid_civic = civic_organic_zealots }
		}
		custom_tooltip = {
			fail_text = "civic_tooltip_overlord_not_organic_zealots_fail"
			NAND = {
				is_subject = yes
				exists = overlord
				overlord = { has_valid_civic = civic_organic_zealots }
			}
		}
	}
	# Unlocks Synthetic Assimilation.
	on_enabled = {
		hidden_effect = {
			country_event = {
				id = utopia.2550
			}
		}
	}
	custom_tooltip = tr_synthetics_synthetic_age_desc
	tradition_swap = {
		name = tr_synthetics_assembly_systems
		inherit_effects = no
		inherit_icon = no
		trigger = {
			has_civic = civic_fanatic_purifiers
			has_technology = tech_synthetic_leaders
		}
		custom_tooltip = tr_synthetics_assembly_systems_desc
		modifier = {
			planet_pop_assembly_mult = 0.10
		}
		on_enabled = {
			hidden_effect = {
				country_event = {
					id = utopia.2550
				}
			}
		}
		weight = {
			factor = 1
		}
	}
	tradition_swap = {
		name = tr_synthetics_power_systems
		inherit_effects = no
		inherit_icon = no
		trigger = {
			is_machine_empire = yes
		}
		modifier = {
			MACHINE_species_trait_picks_add = 1
		}
		weight = {
			factor = 2
		}
	}
}
