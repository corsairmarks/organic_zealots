# this = country (potential owner)
create_oracle_species = {
	if = {
		limit = { NOT = { species_portrait = art13 } }
		create_species = {
			name = NAME_Oracle
			plural = NAME_Oracle
			class = ART
			portrait = art13
			namelist = ART1
			effect = {
				save_global_event_target_as = oracle_species
			}
		}
	}
	else_if = {
		limit = { species_portrait = art13 }
		create_species = {
			name = NAME_Oracle
			plural = NAME_Oracle
			class = ART
			portrait = art17
			namelist = ART1
			effect = {
				save_global_event_target_as = oracle_species
			}
		}
	}
}

	create_saved_leader = {

		gender = female
		immortal = yes

		traits = {
			trait = leader_trait_governor_oracle
		}
		effect = {
			# set_immortal = yes

		}
	}
}
